```csharp
using System.Runtime.InteropServices;
using System.Runtime.Versioning;
using global::Windows.Win32;
using global::Windows.Win32.Foundation;
using global::Windows.Win32.Graphics.Gdi;
using global::Windows.Win32.UI.WindowsAndMessaging;

namespace Focus.Windows.Daemon.Overlay;

/// <summary>
/// Renders mode-indicator arrows on layered overlay windows using direct pixel buffer writes.
/// Move mode: 4 compass arrows at window center.
/// Resize mode: left/right arrow at right edge center, up/down arrow at top edge center.
/// Follows the same DIB + UpdateLayeredWindow pipeline as BorderRenderer.
/// </summary>
[SupportedOSPlatform("windows6.0.6000")]
internal static class ArrowRenderer
{
    // Arrow height in pixels, clamped to reasonable range
    private static int GetArrowSize(int width, int height)
        => Math.Clamp(Math.Min(width, height) / 8, 16, 48);

    public static unsafe void PaintMoveArrows(HWND hwnd, RECT bounds, uint argbColor) { ... }
    public static unsafe void PaintResizeArrows(HWND hwnd, RECT bounds, uint argbColor) { ... }
}
```

**PaintMoveArrows** (OVRL-01): Renders 4 filled triangle arrows arranged as a compass/cross at window center.
- Arrow size from `GetArrowSize(width, height)`. Base width = arrowSize * 2/3.
- Up arrow: tip at `(cx, cy - arrowSize)`, base at `(cx - base/2, cy - gap)` to `(cx + base/2, cy - gap)` where gap = arrowSize/3 (spacing from center)
- Down arrow: tip at `(cx, cy + arrowSize)`, base from `(cx - base/2, cy + gap)` to `(cx + base/2, cy + gap)`
- Left arrow: tip at `(cx - arrowSize, cy)`, base from `(cx - gap, cy - base/2)` to `(cx - gap, cy + base/2)`
- Right arrow: tip at `(cx + arrowSize, cy)`, base from `(cx + gap, cy - base/2)` to `(cx + gap, cy + base/2)`
- cx = width/2, cy = height/2 (DIB-local coordinates, NOT screen coordinates)

**PaintResizeArrows** (OVRL-02, OVRL-03): Renders axis indicators.
- At right edge center: left-pointing triangle + right-pointing triangle (back-to-back pair indicating horizontal axis). Position in DIB: `(width - arrowSize*2, height/2)`
- At top edge center: up-pointing triangle + down-pointing triangle (back-to-back pair indicating vertical axis). Position in DIB: `(width/2, arrowSize*2)`
- Arrow size same as move mode. Each triangle in a pair is slightly smaller (arrowSize * 3/4) so the pair fits.

**Triangle rendering:** For each triangle defined by 3 vertices (x0,y0), (x1,y1), (x2,y2):
1. Compute bounding box
2. For each pixel in bounding box, test if inside triangle using cross-product sign test:
   ```csharp
   static bool IsInsideTriangle(int px, int py, int x0, int y0, int x1, int y1, int x2, int y2)
   {
       int d1 = (px - x2) * (y0 - y2) - (x0 - x2) * (py - y2);
       int d2 = (px - x0) * (y1 - y0) - (x1 - x0) * (py - y0);
       int d3 = (px - x1) * (y2 - y1) - (x2 - x1) * (py - y1);
       bool hasNeg = d1 < 0 || d2 < 0 || d3 < 0;
       bool hasPos = d1 > 0 || d2 > 0 || d3 > 0;
       return !(hasNeg && hasPos);
   }
   ```
3. If inside, write premultiplied ARGB pixel (same formula as BorderRenderer)

**DIB pipeline** (identical to BorderRenderer.PaintFullBorder):
1. `PInvoke.GetDC(HWND.Null)` for screen DC
2. `PInvoke.CreateDIBSection` with 32bpp top-down bitmap
3. `PInvoke.CreateCompatibleDC` + `SelectObject`
4. `GdiFlush` + `NativeMemory.Clear` to zero all pixels
5. Write arrow pixels into buffer
6. `UpdateLayeredWindow` with `ULW_ALPHA` and per-pixel alpha blend
7. Cleanup: `SelectObject(oldBitmap)`, `DeleteDC`, `DeleteObject(hBitmap)`, `ReleaseDC`

The DIB covers the full window bounds (same size as the border overlay). Arrow positions are relative to the DIB origin (0,0 = top-left of window visible bounds).

   {
       if (_shutdownRequested) return;
       _currentMode = mode;  // Set BEFORE Invoke so it's available in the STA callback
       try
       {
           _staDispatcher.Invoke(() =>

   ```csharp
   private unsafe void RefreshForegroundOverlayOnly()
   {
       _overlayManager.HideAll();

       var fgHwnd = PInvoke.GetForegroundWindow();
       if (fgHwnd == default) return;

       RECT fgBounds = default;
       var boundsBytes = MemoryMarshal.AsBytes(MemoryMarshal.CreateSpan(ref fgBounds, 1));
       var hr = PInvoke.DwmGetWindowAttribute(
           fgHwnd,
           DWMWINDOWATTRIBUTE.DWMWA_EXTENDED_FRAME_BOUNDS,
           boundsBytes);

       if (hr.Succeeded && (fgBounds.right - fgBounds.left) > 0)
       {
           // Mode-specific border color
           uint borderColor = _currentMode switch
           {
               WindowMode.Move => MoveModeColor,
               WindowMode.Grow => GrowModeColor,
               _               => ForegroundBorderColor
           };
           _overlayManager.ShowForegroundOverlay(fgBounds, borderColor);

           // Mode-specific arrows (OVRL-01, OVRL-02, OVRL-03)
           if (_currentMode == WindowMode.Move || _currentMode == WindowMode.Grow)
           {
               uint arrowColor = _currentMode == WindowMode.Move ? MoveModeColor : GrowModeColor;
               _overlayManager.ShowModeArrows(fgBounds, _currentMode, arrowColor);
           }
       }
   }
   ```
